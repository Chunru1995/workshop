#CLEAN DATA
#h2a
h2a_2021 <- read_xlsx(here("assignments/02-clean-h2a/", 
                               "H-2A_Disclosure_Data_FY2021.xlsx")) %>%
  filter(CASE_STATUS %in% c("Determination Issued - Certification",

#acs
poverty <- get_acs(geography = "state", table = "B17001", output = "wide") %>%
  select(state = NAME, pov_pop = B17001_001E, pov = B17001_002E) %>%
  mutate(per_pov = (pov / pov_pop) * 100) %>%
  filter(!state %in% c("Alaska", "Hawaii", "District of Columbia", "Puerto Rico"))
## Getting data from the 2017-2021 5-year ACS

race <- get_acs(geography = "state", table = "B03002", output = "wide") %>%

         totpop = B03002_001E, 
         black = B03002_004E,
         aian = B03002_005E,

#residualize variables
pov_mod <- lm(per_pov ~ per_hisp + per_black + per_aian + per_ag, 
              data = state_sf)
